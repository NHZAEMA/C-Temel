{
  float radius;
  float circum, area;

  cout << "Yaricapi yaziniz.";

//Kullanýcýnýn belirttiði iþlemi yapan program:

#include <cstdlib>
#include <iostream>

using namespace std;

int main()
{
int num1, num2, result;
char op;

cout << "Islemi yaziniz.";
cin >> num1 >> op >> num2;
switch (op) {
  case '+': result = num1 + num2; break;
  case '-': result = num1 - num2;break;
  case '*': result = num1 * num2;break;
  case '/': result = num1 / num2;break;
  case '%': result = num1 % num2;break;
  
  default: cout << "Boyle bir islem yok." << endl;
}
cout << num1 << op << num2 << "Ýslemin sonucu:" << result << endl;
return 0;
}
