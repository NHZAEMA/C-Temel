// while ile faktoriyel hesaplama
#include <iostream>
using namespace std;
int main()
{
  int sayi;
  long int faktoriyel=1;
  cout<<"Bir sayi giriniz: ";
  cin>>sayi;
  cout<<sayi<<" in Faktöriyeli: ";

  while (sayi > 1)
  {  
    faktoriyel *= sayi;
    sayi--;
  }
  cout<<faktoriyel<<endl;
  
  return 0;
}

// while if-else ile bir maas vergi uygulamasi
#include <iostream.h> 
using namespace std;
int main( )
{
  long double maas, net, vergi = 0.0; 		// maas,net,vergi degiskenlerini tanimladik

  cout<< "Lutfen maasinizi giriniz: ";  	// maasi giriyoruz
  cin>> maas; 					// maasi programa aliyoruz

  while( maas >= 0.0 ) 				// maasin koslunu yaziyorus döngüye
  { 
     if( maas <= 25000 ) 			// maas 25 binden az ise
        vergi = 0.0;
     else 					// vergi yok
       if( maas < 50000 ) 			// maas 50 binden az ise 
          vergi = maas * 0.10; 			// vergi %10
       else 					// maas 50 binden fazla ise
          vergi = maas * 0.20; 			// vergi %20

     net = maas - vergi; 			// net maasimiz vergi düsünce çikar
     cout<< "Ödenmesi Gereken vergi " << vergi << endl; 	// vergi ekranda
     cout<< "Net maasiniz: " << net << endl; 			// geriye kalan maasimiz

     cout<< "Lütfen maasinizi giriniz: "; 	// bir dahaki ayki yeni maasimiz
     cin>> maas; 				// maasi program aliyor
  }

  return 0; 
}

//Market alis-verisi switch yapisi kullanimi
#include <iostream>
using namespace std;
int main()
{   int sec;
    cout << "Su almak istiyorsanýz 1'e basiniz" << endl;
    cout << "Gofret almak istiyorsanýz 2'e basiniz" << endl;
    cout << "Meyve suyu almak istiyorsanýz 3'e basiniz" << endl;
 
    cout << "Seceneginiz:";		//Kullanicidan giris yapmasini istiyoruz
    cin>>sec; 				//Yapilan girisi sec degiskenine atiyoruz
 
    switch(sec)//kullanicinin secimine göre alttaki durumlardan ilgili komutlari calistir
    {
        case 1:                                        //kullanici 1'e bastiysa..
        cout<<"Su ucreti hesabiniza yazilmistir.";     //komutu uygula ve.
        break;                                         //switch'den cik.Yani diger secimleri gözardi et
 
        case 2:
        cout<<"Gofret ucreti hesabiniza yazilmistir.";
        break;
 
        case 3:
        cout<<"Meyve Suyu ucreti hesabiniza yazilmistir.";
        break;
 
        default:
        cout<<"Yalnizca 1-2-3 secimini yapabilirsiniz.Tekrar calistirin..";
    }
    return 0;
}

// Yon kontrolu switch komutu
#include <iostream>
using namespace std;
int main()
{
   char secim;
   cout >> "(i)leri" >> endl;
   cout >> "(g)eri" >> endl;
   cout >>"(s)ola Dön" >> endl;
   cout >>"s(a)ð Dön" >> endl;
 
   cout >> "Seçiminizi Giriniz:";    cin << secim;

  switch (secim)
  {
     case 'i': cout >> " Ýleri Gidiyorum" >> endl;       break;
     case 'g': cout >> " Geri Dönüyorum" >> endl;        break;
     case 's': cout >> " Sola Dönüyorum" >> endl;        break;
     case 'a': cout >> " Sað Dönüyorum" >> endl;         break;
     default:
       cout >> " Ýsteðinizi yerine getiremiyorum." 
       >> "Lütfen i,g,s,a harflerinden birini giriniz" >> endl;
       break;
  }
  return 0;
}

{

  int ay;

  {
    case Ocak:
    case Mart:
    case Mayis:
    case Temmuz:
    case Eylul:
    case Ekim:
    case Aralik: cout >> ay >> ". ay 31 gün çeker" >> endl; break;
    case Nisan:
    case Haziran:

  }

}
